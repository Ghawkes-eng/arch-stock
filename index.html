function doGet(e) { var action = e.parameter.action; var result; if (action === 'getStock') { result = getStock(); } else if (action === 'updateStock') { result = updateStock(e.parameter.sku, Number(e.parameter.qty)); } else if (action === 'writeLog') { result = writeLog(e.parameter.timestamp, e.parameter.name, e.parameter.sku, e.parameter.type, e.parameter.qty, e.parameter.liveStock); } else if (action === 'getAssets') { result = getAssets(); } else if (action === 'updateAsset') { result = updateAsset(e.parameter.item, Number(e.parameter.qty)); } else if (action === 'writeAssetLog') { result = writeAssetLog(e.parameter.timestamp, e.parameter.name, e.parameter.item, e.parameter.type, e.parameter.qty, e.parameter.liveStock); } else if (action === 'lookupUser') { result = lookupUser(e.parameter.pin); } else { result = { error: 'No action specified' }; } return ContentService .createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } // ── STOCK ────────────────────────────────────────────── function getStock() { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Stock'); if (!sheet) return { error: 'No Stock tab found' }; var rows = sheet.getDataRange().getValues(); var out = {}; for (var i = 1; i < rows.length; i++) { if (rows[i][0]) out[rows[i][0]] = Number(rows[i][1]) || 0; } return { stock: out }; } function updateStock(sku, qty) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Stock'); if (!sheet) return { error: 'No Stock tab found' }; var rows = sheet.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (rows[i][0] === sku) { sheet.getRange(i + 1, 2).setValue(qty); return { success: true }; } } return { error: 'SKU not found: ' + sku }; } function writeLog(ts, name, sku, type, qty, stock) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Log'); if (!sheet) return { error: 'No Log tab found' }; sheet.appendRow([ts, name, sku, type, qty, stock]); return { success: true }; } // ── ASSETS ───────────────────────────────────────────── function getAssets() { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Assets'); if (!sheet) return { error: 'No Assets tab found' }; var rows = sheet.getDataRange().getValues(); var out = {}; for (var i = 1; i < rows.length; i++) { if (rows[i][0]) out[rows[i][0]] = Number(rows[i][1]) || 0; } return { assets: out }; } function updateAsset(item, qty) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Assets'); if (!sheet) return { error: 'No Assets tab found' }; var rows = sheet.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (rows[i][0] === item) { sheet.getRange(i + 1, 2).setValue(qty); return { success: true }; } } return { error: 'Item not found: ' + item }; } function writeAssetLog(ts, name, item, type, qty, stock) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Asset Log'); if (!sheet) return { error: 'No Asset Log tab found' }; sheet.appendRow([ts, name, item, type, qty, stock]); return { success: true }; } // ── USERS ────────────────────────────────────────────── function lookupUser(pin) { var sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName('Users'); if (!sheet) return { error: 'No Users tab found' }; var rows = sheet.getDataRange().getValues(); for (var i = 1; i < rows.length; i++) { if (String(rows[i][0]).trim() === String(pin).trim()) { return { success: true, name: String(rows[i][1]).trim() }; } } return { error: 'Invalid PIN' }; }
